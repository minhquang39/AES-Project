function Person(name, age, address) {
this.name = name;
this.age = age;
this.address = address;
this.getInfo = function () {
return `${this.name}${this.age}`;
};
}
Person.prototype.class = "F8";
Person.prototype.getClass = function () {
return this.class;
};
var user = new Person("Minh Quang", 21, "Nghe An");
var user2 = new Person("Huy Quan", 21, "Ha Noi");
console.log(Person);
console.log(user);
console.log(user2);
console.log(user2.getClass());
